\section{shared\_bundle 交付模型}
\subsection{目录位置}
\begin{itemize}
  \item VS/MSVC：\texttt{build/newdb-vs2026-mt-tests/shared\_bundle/Release}
  \item MinGW：\texttt{build/newdb-mingw/shared\_bundle/Release}
  \item WSL/Linux：\texttt{build/newdb-wsl-tests/shared\_bundle/Release}
\end{itemize}

\subsection{构建命令}
\begin{lstlisting}
# VS/MSVC
cmake --build build/newdb-vs2026-mt-tests --config Release --target shared_bundle --parallel

# MinGW
cmake --build build/newdb-mingw --target shared_bundle --parallel

# WSL/Linux
cmake --build build/newdb-wsl-tests --target shared_bundle -j8
\end{lstlisting}

\subsection{依赖优先策略}
\begin{itemize}
  \item Windows：使用 \texttt{run\_*.cmd}，优先加载 bundle 目录内 DLL。
  \item Linux/WSL：通过 \texttt{RPATH=\$ORIGIN}，优先加载同目录 \texttt{.so}。
  \item MinGW：自动附带 \texttt{libstdc++-6.dll}、\texttt{libgcc\_s\_*.dll}、
  \texttt{libwinpthread-1.dll}。
\end{itemize}

\subsection{错误码}
\begin{longtable}{>{\ttfamily}p{0.36\linewidth}p{0.56\linewidth}}
\toprule
错误码 & 含义 \\
\midrule
NEWDB\_OK & 成功 \\
NEWDB\_ERR\_INVALID\_ARGUMENT & 参数错误/缓冲区契约错误 \\
NEWDB\_ERR\_INVALID\_HANDLE & 会话句柄无效或已销毁 \\
NEWDB\_ERR\_EXECUTION\_FAILED & 命令或业务执行失败 \\
NEWDB\_ERR\_INTERNAL & 内部错误 \\
NEWDB\_ERR\_LOG\_IO & 日志/文件 I/O 失败 \\
NEWDB\_ERR\_SESSION\_TERMINATED & 会话已终止不可继续 \\
\bottomrule
\end{longtable}

\subsection{线程安全约定}
\begin{itemize}
  \item 全局元信息接口线程安全。
  \item 会话接口按“一线程一 handle”安全。
  \item 同一 handle 并发调用不保证安全。
\end{itemize}

\section{结果解析规范}
必须同时判断三项：
\begin{enumerate}
  \item \texttt{rc}（\texttt{newdb\_session\_execute} 返回值）
  \item \texttt{newdb\_session\_last\_error}
  \item \texttt{output\_buf}（可读诊断）
\end{enumerate}

建议统一结果结构：
\begin{lstlisting}
ExecuteResult {
  rc: int
  last_error: int
  error_name: string
  output: string
  ok: bool
}
\end{lstlisting}

缓冲区建议：
\begin{itemize}
  \item 常规命令至少 4KB
  \item 复杂查询建议 16KB 或以上
\end{itemize}

\section{部署检查清单}
\begin{enumerate}
  \item 必须整目录分发 \texttt{shared\_bundle/Release}
  \item 必须绝对路径加载主库
  \item 首次调用前必须做 ABI 协商
  \item 必须采用一线程一会话句柄
  \item 每次调用记录 \texttt{rc + last\_error + output}
\end{enumerate}

\section{常见问题}
\begin{itemize}
  \item 找不到模块：通常缺少同目录依赖。
  \item 加载到系统同名库：请使用绝对路径和 launcher。
  \item WSL 拉 gtest 失败：使用本地
  \texttt{FETCHCONTENT\_SOURCE\_DIR\_GOOGLETEST}。
\end{itemize}
